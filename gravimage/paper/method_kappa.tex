\begin{equation}
    \label{eqn:kappa}
    \langle{v_{\rm LOS}^4}\rangle = \frac{2}{\Sigma(R)}\int_R^\infty
    \frac{\nu \langle {v_r^4}\rangle r}{\sqrt{r^2-R^2}}g(r,R,\beta')\text{d}r
\end{equation}
\begin{equation*}
    \label{eqn:g}
    g(r,R,\beta') = 1-2\beta'\frac{R^2}{r^2}+\frac{\beta'(1+\beta')}{2}\frac{R^4}{r^4},
\end{equation*}

\begin{equation*}

\end{equation*}
