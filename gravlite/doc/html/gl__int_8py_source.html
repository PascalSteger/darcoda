00073 ## function from Lokas+2005, eq. (8)
00074 # @param rvar small r, in [pc]

00076 # @param beta beta, in [1]
00077 # @param dbetadr d beta/dr, in [1]
00078 def g(rvar, rfix, beta, dbetadr):
00079     tmp = 1.             # [1]
00080     tmp -= 2*beta*rfix**2/rvar**2       # [1]
00081     tmp += beta*(beta+1.)/2.*rfix**4/rvar**4 # [1]

00083     gh.checknan(tmp)
