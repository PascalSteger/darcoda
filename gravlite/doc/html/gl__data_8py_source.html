00125     def read_mass(self):

00127         # [rcore], [totmass], [totmass]

00129         # switch to Munit (msun) and pc here

00131         self.binmin     = self.binmin[:]  * gp.rcore_2D[0]
00132         self.binmax     = self.binmax[:]  * gp.rcore_2D[0]
00133         self.Mdat_2D    = self.Mdat_2D[:] * gp.totmass[0]      # [munit/pc**2]
00134         self.Merr_2D    = self.Merr_2D[:] * gp.totmass[0]      # [munit/pc**2]
00135         
00136         # calculate surface density 
00137         self.densx_2D   = self.Mx_2D       # [pc]
00138         # use binmax here, as mass is enclosed withing max radius of bin!
00139         self.densdat_2D = phys.calculate_surfdens(self.binmax, self.Mdat_2D) # [munit/pc**2]
00140         self.denserr_2D = self.densdat_2D * self.Merr_2D/self.Mdat_2D # [munit/pc**2]
00141 
00142         # deproject, already normalized to same total mass
00143         # possibility: read density file, is not possible as this is only 2D density
00144         self.densx, self.densdat, self.denserr = Rho_NORM_rho(self.densx_2D[:],self.densdat_2D[:],self.denserr_2D[:])
00145         # takes [pc], 2* [munit/pc**2], gives [pc], 2*[munit/pc**3]
00146         
00147         # needed for baryonic prior
00148         self.Mx   = self.densx[:]                       # [pc,3D]
00149         self.Mdat = rho_SUM_Mr(self.densx, self.densdat) # [Munit], 3D
