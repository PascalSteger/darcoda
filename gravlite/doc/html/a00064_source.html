135 def ipol_rhalf_log(X, Y, rhalf):
136  for i in range(len(X)):
137  if X[i] < rhalf:
138  lowr = X[i]
139  lowi = i
140  if X[len(X)-i-1] > rhalf:
141  highr = X[len(X)-i-1]
142  highi = len(X)-i-1
143  dr = np.log(highr) - np.log(lowr)
144  dy = np.log(Y[highi]) - np.log(Y[lowi])
145 
146  Yhalf = np.exp(np.log(Y[lowi]) + (np.log(rhalf)-np.log(lowr))*dy/dr)
147  return Yhalf
